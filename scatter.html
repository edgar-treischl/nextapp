Interactive Scatter Plot Example

Code Example

const data = {
  datasets: [{
    label: 'Dataset 1',
    data: [
      { x: 10, y: 20 },
      { x: 15, y: 25 },
      { x: 20, y: 30 },
      { x: 25, y: 35 },
      { x: 30, y: 40 },
      { x: 35, y: 45 }
    ],
    backgroundColor: 'rgba(37, 99, 235, 0.7)'
  }]
};

const options = {
  responsive: true,
  plugins: {
    legend: { position: 'top' },
    title: { display: true, text: 'Scatter Plot Example' }
  }
};

<Scatter data={data} options={options} />;
